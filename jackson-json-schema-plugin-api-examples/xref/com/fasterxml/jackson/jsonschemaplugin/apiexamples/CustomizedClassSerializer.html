15  package com.fasterxml.jackson.jsonschemaplugin.apiexamples;
16  
17  import com.fasterxml.jackson.core.JsonGenerator;
18  import com.fasterxml.jackson.databind.SerializerProvider;
19  import com.fasterxml.jackson.databind.ser.std.StdSerializer;
20  
21  import java.io.IOException;
22  
23  /**
24   * Quirky serializer to test schema.
25   */
26  public class CustomizedClassSerializer extends StdSerializer<CustomizedClass> {
27      protected CustomizedClassSerializer() {
28          super(CustomizedClass.class);
29      }
30  
31  
32      @Override
33      public void serialize(CustomizedClass value, JsonGenerator jgen, SerializerProvider provider) throws IOException {
34          jgen.writeStartObject();
35          jgen.writeArrayFieldStart("intValue");
36          jgen.writeNumber(value.getIntvalue());
37          jgen.writeEndArray();
38          jgen.writeEndObject();
39      }
