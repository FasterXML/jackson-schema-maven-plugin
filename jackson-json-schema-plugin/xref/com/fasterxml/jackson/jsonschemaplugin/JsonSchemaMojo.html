31  @Mojo(name = "generate", defaultPhase = LifecyclePhase.PROCESS_CLASSES, requiresDependencyCollection = ResolutionScope.COMPILE, requiresDependencyResolution = ResolutionScope.COMPILE,
32          requiresProject = true)
33  public class JsonSchemaMojo extends AbstractMojo {
34      /**
35       * Location of the result schema.
36       */
37      @Parameter(defaultValue = "${project.build.directory}/schema.json", property = "schema")
38      File outputSchema;
39  
40      /**
41       * Patterns (ant-ish) of classes to generate.
42       * This may not be empty; there is no default.
43       */
44      @Parameter(required = true)
45      List<String> includes;
46  
47      /**
48       * Patterns of classes to exclude.
49       */
50      @Parameter
51      List<String> excludes = Lists.newArrayList(); // default to an empty list instead of null.
52  
53      @Component
54      MavenProject project;
55  
56      public void execute() throws MojoExecutionException, MojoFailureException {
57          ObjectMapper m = new ObjectMapper();
58          SchemaFactoryWrapper visitor = new SchemaFactoryWrapper();
59          try {
60              for (Class<?> clazz : getClassesToProcess()) {
61                  m.acceptJsonFormatVisitor(m.constructType(clazz), visitor);
62              }
63          } catch (IOException e) {
64              throw new MojoFailureException("Failed to construct compile classpath class loader.", e);
65          } catch (DependencyResolutionRequiredException e) {
66              throw new MojoFailureException("Failed to resolve dependencies.", e);
67          }
68          JsonSchema jsonSchema = visitor.finalSchema();
69          try {
70              m.writerWithDefaultPrettyPrinter().writeValue(outputSchema, jsonSchema);
71          } catch (IOException e) {
72              throw new MojoExecutionException("Failed to write result schema.", e);
73          }
74      }
75  
76      private List<Class<?>> getClassesToProcess() throws IOException, DependencyResolutionRequiredException {
77          ClassLoader loader = getClassLoader(); // get the compile classpath class loader.
78          List<Class<?>> results = Lists.newArrayList();
79          ClassPath classPath = ClassPath.from(loader);
80          for (ClassPath.ClassInfo info : classPath.getAllClasses()) {
81              getLog().debug("Class: " + info.getName());
82              boolean included = false;
83              for (String pattern : includes) {
84                  pattern = pattern.replace("/", "."); // map from typical / syntax to class name.
85                  if (SelectorUtils.matchPath(pattern, info.getName(), ".", true)) {
86                      included = true;
87                      break;
88                  }
89              }
90              if (included) {
91                  boolean excluded = false;
92                  for (String pattern : excludes) {
93                      pattern = pattern.replace("/", "."); // map from typical / syntax to class name.
94                      if (SelectorUtils.matchPath(pattern, info.getName(), ".", true)) {
95                          excluded = true;
96                          break;
97                      }
98                  }
99                  if (!excluded) {
100                     results.add(info.load());
101                 }
102             }
103         }
104         return results;
105     }

107     protected ClassLoader getClassLoader() throws DependencyResolutionRequiredException, MalformedURLException {
108         List<URL> urls = new ArrayList<URL>();
109         for (String path : project.getCompileClasspathElements()) {
110                 urls.add(new File(path).toURI().toURL());
111             }
112         return new URLClassLoader(urls.toArray(new URL[urls.size()]));
113     }
